── Access ──────────────────────────────────────────────────
{{- if .Values.ingress.enabled }}
{{- range .Values.ingress.hosts }}
  {{ if $.Values.ingress.tls }}https{{ else }}http{{ end }}://{{ .host }}
{{- end }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc -n {{ .Release.Namespace }} {{ include "kwakeup.fullname" . }} \
    -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes \
    -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "http://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.service.type "LoadBalancer" }}
  (Wait a minute for the cloud load-balancer IP to be assigned, then:)
  export LB_IP=$(kubectl get svc -n {{ .Release.Namespace }} {{ include "kwakeup.fullname" . }} \
    -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "http://$LB_IP:{{ .Values.service.port }}"
{{- else }}
  kubectl port-forward -n {{ .Release.Namespace }} \
    svc/{{ include "kwakeup.fullname" . }} 8080:{{ .Values.service.port }}
  Open: http://localhost:8080
{{- end }}
